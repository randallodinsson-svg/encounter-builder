// ------------------------------------------------------------ // APEXSIM UI CONTROL PANEL — ONE AND DONE // ------------------------------------------------------------ // Wait until APEXSIM exists function bindApexSimControls() { if (!window.APEXSIM) { requestAnimationFrame(bindApexSimControls); return; } // Glow toggle const glowToggle = document.getElementById("sim-toggle-glow"); if (glowToggle) { glowToggle.addEventListener("change", (e) => { APEXSIM.showGlow = e.target.checked; }); } // Trails toggle const trailsToggle = document.getElementById("sim-toggle-trails"); if (trailsToggle) { trailsToggle.addEventListener("change", (e) => { APEXSIM.showTrails = e.target.checked; }); } // Debug vectors toggle const debugToggle = document.getElementById("sim-toggle-debug"); if (debugToggle) { debugToggle.addEventListener("change", (e) => { APEXSIM.showDebugVectors = e.target.checked; }); } } bindApexSimControls();
